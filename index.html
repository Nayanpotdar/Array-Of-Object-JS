            let studentRecords = [ {namr: 'john' , id: 123, marks: 98},

Q2 Suppose we have the same dataset as above but this time we want to get the details of students who scored more than 50 marks. [{name: 'John', id: 123, marks: 98 },{name: 'Wick', id: 115, marks: 75 }]


let studentsWithMoreThan50Marks = studentRecords.filter(student => student.marks > 50);

            console.log(studentsWithMoreThan50Marks);

Q3 Retrieve the details of students who scored more than 50 marks and have IDs greater than 120.


let studentsWithMoreThan50MarksAndIDGreaterThan120 = studentRecords.filter(student => {
        return student.marks > 50 && student.id > 120;
      });
      
      console.log(studentsWithMoreThan50MarksAndIDGreaterThan120);

Q4 Consider the same scenario we have discussed above, but this time we would like to know the sum total of the marks of the students


let totalMarks = studentRecords.reduce((sum, student) => sum + student.marks, 0);

            console.log(totalMarks);

Q5 This time we want to get only the names of the students who scored more than 50 marks from the same dataset used above


let studentsWithMoreThan50MarksNames = studentRecords
            .filter(student => student.marks > 50)

          console.log(studentsWithMoreThan50MarksNames);

Q6 This time we are required to print the sum of marks of the students with id > 120.


let sumOfMarksForIDsGreaterThan120 = studentRecords
        .filter(student => student.id > 120)
        .reduce((sum, student) => sum + student.marks, 0);
      
      console.log(sumOfMarksForIDsGreaterThan120);

Q7 This time we are required to print the total marks of the students with marks greater than 50 after a grace of 15 marks has been added to those students who scored less than 50.


let studentsWithGrace = studentRecords.map(student => ({
            ...student,
            marks: student.marks < 50 ? student.marks + 15 : student.marks
          }));

          let totalMarksAfterGrace = studentsWithGrace
          .filter(student => student.marks > 50)
          .reduce((sum, student) => sum + student.marks, 0);

           console.log(totalMarksAfterGrace);
